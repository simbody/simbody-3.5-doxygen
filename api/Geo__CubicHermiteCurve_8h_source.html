211 static Vec<4,Vec3P> calcAFromH(const Vec<4,Vec3P>& H) {
212  const Vec3P& h0= H[0]; const Vec3P& h1= H[1]; // aliases for beauty
213  const Vec3P& hu0=H[2]; const Vec3P& hu1=H[3];
214  const Vec3P h01 = h0 - h1;

218  h0 );
219 }
